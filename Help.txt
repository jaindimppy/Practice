using Microsoft.AspNetCore.Mvc;
using YourNamespace.Data;
using YourNamespace.DTOs;
using YourNamespace.Models;

namespace YourNamespace.Controllers
{
    [Route("api/[controller]")]
    [ApiController]
    public class AuthController : ControllerBase
    {
        private readonly ApplicationDbContext _context;

        public AuthController(ApplicationDbContext context)
        {
            _context = context;
        }

        [HttpPost("register")]
        public async Task<IActionResult> Register([FromBody] RegisterRequest request)
        {
            if (string.IsNullOrWhiteSpace(request.Email))
                return BadRequest("Email is required");

            var existing = await _context.UserRegistrations
                .FirstOrDefaultAsync(u => u.Email == request.Email && !u.IsVerified);

            if (existing != null)
                return BadRequest("Email already registered but not verified");

            var otp = new Random().Next(100000, 999999).ToString();
            var token = Guid.NewGuid().ToString();

            var registration = new UserRegistration
            {
                Email = request.Email,
                OTP = otp,
                Token = token,
                ExpiresAt = DateTime.UtcNow.AddMinutes(15)
            };

            _context.UserRegistrations.Add(registration);
            await _context.SaveChangesAsync();

            // TODO: Send OTP via email

            return Ok(new
            {
                message = "Registration started. OTP sent to email.",
                registrationId = registration.RegistrationId
            });
        }
    }
}

using System.Net;
using System.Text.Json;

public class GlobalExceptionMiddleware
{
    private readonly RequestDelegate _next;
    private readonly ILogger<GlobalExceptionMiddleware> _logger;

    public GlobalExceptionMiddleware(RequestDelegate next, ILogger<GlobalExceptionMiddleware> logger)
    {
        _next = next;
        _logger = logger;
    }

    public async Task InvokeAsync(HttpContext context)
    {
        try
        {
            await _next(context); // Continue pipeline
        }
        catch (Exception ex)
        {
            _logger.LogError(ex, "An unhandled exception occurred.");

            context.Response.ContentType = "application/json";
            context.Response.StatusCode = (int)HttpStatusCode.InternalServerError;

            var errorResponse = new
            {
                StatusCode = context.Response.StatusCode,
                Message = "Something went wrong. Please try again later.",
                Error = ex.Message // (optional: hide in production)
            };

            var result = JsonSerializer.Serialize(errorResponse);
            await context.Response.WriteAsync(result);
        }
    }
}

Ardalis Specification
----------------------------------------
dotnet add package Ardalis.Specification.EntityFrameworkCore


Ardalis Specification is mainly for separating and reusing the "where" (filter) or query logic that you want to apply on your models/entities.

Why is that useful?
Instead of repeating the same filtering code everywhere (like .Where(u => u.IsActive) in many places),

You put that logic once inside a Specification class,

And then reuse that class wherever you want to query active users.

public class ActiveUsersSpecification : Specification<User>
{
    public ActiveUsersSpecification()
    {
        Query.Where(u => u.IsActive);  // filter: only active users
    }
}

and use it in method like this..

 public async Task<List<User>> GetActiveUsersAsync()

        var spec = new ActiveUsersSpecification();
        return await _userRepository.ListAsync(spec);
